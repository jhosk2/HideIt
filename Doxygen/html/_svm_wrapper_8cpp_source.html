95 double SvmWrapper::predict(const std::vector<std::string>& input, FUNC func)
96 {
97  if ( false == init() )
98  {
99  AppLog("svm init fail");
100  return -1;
101  }
102 
103  int correct = 0;
104  int total = 0;
105  double error = 0;
106  double sump = 0, sumt = 0, sumpp = 0, sumtt = 0, sumpt = 0;
107 
108  int svm_type=svm_get_svm_type(model);
109  int nr_class=svm_get_nr_class(model);
110 
111  double *prob_estimates=NULL;
112  int j;
113 
114  if(predict_probability)
115  {
116  if (svm_type==NU_SVR || svm_type==EPSILON_SVR)
117  {
118  func("Prob. model for test data: target value = predicted value + z,\nz: Laplace distribution e^(-|z|/sigma)/(2sigma),sigma=%g\n",svm_get_svr_probability(model));
119  }
120  else
121  {
122  int *labels=(int *) malloc(nr_class*sizeof(int));
123  svm_get_labels(model,labels);
124  prob_estimates = (double *) malloc(nr_class*sizeof(double));
125  func("labels");
126  for(j=0;j<nr_class;j++)
127  func(" %d",labels[j]);
128  func("\n");
129  free(labels);
130  }
131  }
132 
133  max_line_len = 1024;
134 
135  double predict_label=0;
136  for(std::vector<std::string>::const_iterator itr = input.begin();itr!=input.end();++itr )
137  {
138  char line[1024] ={0,};
139  strcpy( line, (*itr).c_str());
140 
141  int i = 0;
142  double target_label;
143  char *idx, *val, *label, *endptr;
144  int inst_max_index = -1; // strtol gives 0 if wrong format, and precomputed kernel has <index> start from 0
145 
146  label = strtok(line," \t\n");
147  if(label == NULL) // empty line
148  func("Wrong input format at line %d\n", total+1);
149 
150  target_label = strtod(label,&endptr);
151  if(endptr == label || *endptr != '\0')
152  func("Wrong input format at line %d\n", total+1);
153 
154  while(1)
155  {
156  if(i>=max_nr_attr-1) // need one more for index = -1
157  {
158  max_nr_attr *= 2;
159  x = (struct svm_node *) realloc(x,max_nr_attr*sizeof(struct svm_node));
160  }
161 
162  idx = strtok(NULL,":");
163  val = strtok(NULL," \t");
164 
165  if(val == NULL)
166  break;
167  errno = 0;
168  x[i].index = (int) strtol(idx,&endptr,10);
169  if(endptr == idx || errno != 0 || *endptr != '\0' || x[i].index <= inst_max_index)
170  func("Wrong input format at line %d\n", total+1);
171  else
172  inst_max_index = x[i].index;
173 
174  errno = 0;
175  x[i].value = strtod(val,&endptr);
176  if(endptr == val || errno != 0 || (*endptr != '\0' && !isspace(*endptr)))
177  func("Wrong input format at line %d\n", total+1);
178 
179  ++i;
180  }
181  x[i].index = -1;
182 
183  if (predict_probability && (svm_type==C_SVC || svm_type==NU_SVC))
184  {
185  predict_label = svm_predict_probability(model,x,prob_estimates);
186  func("%g",predict_label);
187  for(j=0;j<nr_class;j++)
188  func(" %g",prob_estimates[j]);
189  func("\n");
190  }
191  else
192  {
193  predict_label = svm_predict(model,x);
194  func("%g\n",predict_label);
195  }
196 
197  if(predict_label == target_label)
198  ++correct;
199  error += (predict_label-target_label)*(predict_label-target_label);
200  sump += predict_label;
201  sumt += target_label;
202  sumpp += predict_label*predict_label;
203  sumtt += target_label*target_label;
204  sumpt += predict_label*target_label;
205  ++total;
206 
207  }
208 
209  if (svm_type==NU_SVR || svm_type==EPSILON_SVR)
210  {
211  func("Mean squared error = %g (regression)\n",error/total);
212  func("Squared correlation coefficient = %g (regression)\n",
213  ((total*sumpt-sump*sumt)*(total*sumpt-sump*sumt))/
214  ((total*sumpp-sump*sump)*(total*sumtt-sumt*sumt))
215  );
216  }
217  else
218  func("Accuracy = %g%% (%d/%d) (classification)\n",
219  (double)correct/total*100,correct,total);
220  if(predict_probability)
221  free(prob_estimates);
222 
223  svm_free_and_destroy_model(&model);
224  free(x);
225 
226  return predict_label;
227 }
