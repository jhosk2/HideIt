173  m_nTop = 0;
174  m_nStack = new int[MAX];
175  m_strPostfix = new char[MAX];
176 
177  // 연산 순서에 맞게 배치한다.
178  int i = 0, j = 0, nCount = 0;
179  while (m_strInfix[i]) {
180  // 괄호 처리
181  if (m_strInfix[i] == '(')
182  PushStack(m_strInfix[i++]);
183  else if (m_strInfix[i] == ')') {
184  while (GetStack() != '(')
185  m_strPostfix[j++] = PopStack();

191  PopStack();
192  i++;
193  }
194 
195  // 연산자 처리
196  else if (IsOperator(m_strInfix[i])) {
197  // 부호 연산자 처리
198  int flag = (nCount == 0) || m_strInfix[i - 1] == '('

200  if (m_strInfix[i] == '-' && flag) {
201  // 마이너스 숫자일 때 앞에 ~ 넣어줌

204  }

207  i++;
208  //*/
209  else {
210  while (!IsEmpty()
211  && GetOrder(GetStack()) >= GetOrder(m_strInfix[i]))
212  m_strPostfix[j++] = PopStack();
213  PushStack(m_strInfix[i++]);
214  }
215  }
216  // 숫자 처리
217  else if (isdigit(m_strInfix[i]) || (m_strInfix[i] == '.')) {
218  while (isdigit(m_strInfix[i]) || (m_strInfix[i] == '.'))
219  m_strPostfix[j++] = m_strInfix[i++];
220  m_strPostfix[j++] = ' ';

224  if (isalpha(m_strInfix[i + 1])) {
225  char* temp = new char[20];
226  int k = 0;
227  while (isalpha(m_strInfix[i]))
228  temp[k++] = m_strInfix[i++];
229  temp[k] = '\0';
230 
231  if (int code = IsFunction(temp))
232  PushStack(code);

237  m_strPostfix[j++] = m_strInfix[i++];
238  } else
239  i++;
240  nCount++;
241  }
242 
243  while (!IsEmpty())
244  m_strPostfix[j++] = PopStack();
245  m_strPostfix[j++] = '\0';
246  return;
247 }

249 bool Calculate::ResultRangeCheck(double value_1, double value_2, char exp) {
250  double double_max = std::numeric_limits<double>::max();
251  double double_min = std::numeric_limits<double>::min();
252 
253  switch (exp) {
254  case '^': {
255  if (pow(value_1, value_2) == 0)
256  return true;
257  else {
258  if (pow(value_1, value_2) > double_max
259  || pow(value_1, value_2) < double_min)
260  return false;
261  else
262  return true;
263  }
264  }
265  case '+': {
266  if ((value_1 + value_2) == 0)
267  return true;
268  else {
269  if ((value_1 + value_2) > double_max
270  || (value_1 + value_2) < double_min)
271  return false;
272  else
273  return true;
274  }
275  }
276  case '-': {
277  if ((value_1 - value_2) == 0)
278  return true;
279  else {
280  if ((value_1 - value_2) > double_max
281  && (value_1 - value_2) < double_min) {
282  return false;
283  } else
284  return true;
285  }
286  }
287  case '*': {
288  if ((value_1 * value_2) == 0)
289  return true;
290  else {
291  if ((value_1 * value_2) > double_max
292  || (value_1 * value_2) < double_min)
293  return false;
294  else
295  return true;
296  }
297  }
298  case '/': {
299  if ((value_1 / value_2) == 0)
300  return true;
301  else {
302  if ((value_1 / value_2) > double_max
303  || (value_1 / value_2) < double_min)
304  return false;
305  else
306  return true;
307  }
308  }
309  }
