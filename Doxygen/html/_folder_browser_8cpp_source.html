148 void FolderBrowser::GotoDirectory(const String& targetFolder) {
149 
150  currentLocation = targetFolder;
151  HidecurrentLocation = targetFolder;
152 
153  Directory* pDir = new Directory; // Allocates the %Directory instance
154 
155  // Open the directory
156  result r = pDir->Construct(currentLocation);
157  if (IsFailed(r)) {
158  switch (r) {
159  case E_SUCCESS:
160  break;
161  case E_INVALID_ARG:
162  break;
163  case E_ILLEGAL_ACCESS:
164  break;
165  case E_FILE_NOT_FOUND:
166  break;
167  case E_MAX_EXCEEDED:
168  break;
169  case E_IO:
170  break;
171  }
172  return;
173  }
174 
175  // Reads all the directory entries
176  DirEnumerator* pDirEnum = pDir->ReadN();
177  if (!pDirEnum) {
178  return;
179  }
180  items.RemoveAll(true);
181  // Loops through all the directory entries
182  while (pDirEnum->MoveNext() == E_SUCCESS) {
183  DirEntry entry = pDirEnum->GetCurrentDirEntry();
184  if (entry.GetName().StartsWith(L".", 0))
185  continue;
186  if ( IsFiltered(entry))
187  items.Add(*(new DirEntry(entry)));
188  }
189 
190  // Releases the %Directory instance.
191  // The opened directory is closed automatically when the destructor of the %Directory class is invoked.
192 
193  delete pDirEnum;
194 
195  delete pDir;
196  pDir = null;
197 
198  if (pLabel != null){
199  pLabel->SetText(currentLocation);
200  pLabel->Invalidate(true);
201  }
202  listView->UpdateList();
203 }

228 
229  Tizen::Graphics::Bitmap* bitmap;
230 
231  if (attr.IsDirectory()) {
232  bitmap = pBitmapFolder;
233  } else {
234 
235  String ext = File::GetFileExtension(fullPath);
236  if (ext.Equals(L"png", false) || ext.Equals(L"bmp", false)
237  || ext.Equals(L"jpg", false) || ext.Equals(L"jpeg", false)
238  || ext.Equals(L"gif", false)) {
239  bitmap = pBitmapImage;
240  } else if (ext.Equals(L"txt", false) || ext.Equals(L"doc", false)
241  || ext.Equals(L"docx", false) || ext.Equals(L"hwp", false)) {
242  bitmap = pBitmapText;
243  } else if (ext.Equals(L"mp3", false) || ext.Equals(L"wav", false)
244  || ext.Equals(L"ogg", false)) {
245  bitmap = pBitmapSound;
246  } else if (ext.Equals(L"mp4", false) || ext.Equals(L"mkv", false)
247  || ext.Equals(L"avi", false)) {
248  bitmap = pBitmapVideo;
249  } else {
250  bitmap = pBitmapMisc;
251  }
252  }
253 
254  pIconListviewItem->Construct(*bitmap, &fileName);
255  AppLog("CreateeItem Start OK");

304 bool FolderBrowser::IsFiltered( const DirEntry& entry ){
305 
306  String ext = File::GetFileExtension(entry.GetName());
307 
308  switch(fileType)
309  {
310  case H_IMAGE:
311  if (ext.Equals(L"png", false) || ext.Equals(L"bmp", false)
312  || ext.Equals(L"jpg", false) || ext.Equals(L"jpeg", false)
313  || ext.Equals(L"gif", false))
314  return true;
315  break;
316  case H_VIDEO:
317  if (ext.Equals(L"mp4", false) || ext.Equals(L"mkv", false)
318  || ext.Equals(L"avi", false))
319  return true;
320  break;
321  case H_ETC:
322  if (ext.Equals(L"png", false) || ext.Equals(L"bmp", false)
323  || ext.Equals(L"jpg", false) || ext.Equals(L"jpeg", false)
324  || ext.Equals(L"gif", false))
325  return false;
326 
327  if (ext.Equals(L"mp4", false) || ext.Equals(L"mkv", false)
328  || ext.Equals(L"avi", false))
329  return false;
330  return true;
331  break;
332  case H_ALL:
333  return true;
334  break;
335  }
336  return false;
337 }
