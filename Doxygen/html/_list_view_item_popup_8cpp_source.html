108 void ListViewItemPopup::OnActionPerformed(const Control& source, int actionId) {
109 
110  switch (actionId) {
111  case ID_BUTTON_OPEN_POPUP: {
112 
113  String targetFile;
114  if (FILE_STATE_FLAG != FILESTATE_DECRYPTION) {
115  Cryptor cryptor;
116  cryptor.Decrypt(currentFilePath, true);
117 
118  targetFile.Append(TEMPPATH);
119  targetFile.Append(File::GetFileExtension(currentFilePath));
120  }
121  else
122  {
123  targetFile = currentFilePath;
124  }
125 
126  String ext = File::GetFileExtension(targetFile);
127  if (ext.Equals(L"txt", false) || ext.Equals(L"doc", false)
128  || ext.Equals(L"docx", false) || ext.Equals(L"hwp", false)) {
129  ExecuteTextFile(targetFile);
130  }
131  else
132  {
133  result r;
134  try{
135  r = Tizen::App::AppControl::FindAndStart(L"http://tizen.org/appcontrol/operation/view", &targetFile, null, null, null, null);
136  }
137  catch(result r2)
138  {
139 
140  }
141  switch(r)
142  {
143  case E_SUCCESS:
144  break;
145  case E_MAX_EXCEEDED:
146  break;
147  case E_OBJ_NOT_FOUND:
148  break;
149  case E_PRIVILEGE_DENIED:
150  break;
151  case E_SYSTEM:
152  break;
153  case E_ILLEGAL_ACCESS:
154  break;
155  }
156  }
157  HidePopup();
158 
159  if (FILE_STATE_FLAG != FILESTATE_DECRYPTION) {
160  //File::Remove(targetFile);
161  }
162  }
163 
164  break;
165 
166  case ID_BUTTON_CLOSE_POPUP: {
167  HidePopup();
168  }
169  break;
170 
171  case ID_BUTTON_ENCRYPTION_FILE :
172  {
173  Cryptor cryptor;
174 
175  if (FILE_STATE_FLAG == FILESTATE_DECRYPTION) {
176  cryptor.Encrypt(currentFilePath);
177  File::Remove(currentFilePath);
178  }
179  else
180  {
181  cryptor.Decrypt(currentFilePath);
182  File::Remove(currentFilePath);
183  }
184  HidePopup();
185  //TODO file encryption, 플래그 변경
186  }
187  break;
188  default:
189  break;
190  }
